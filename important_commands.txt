## starting frontend UI server
npm start

## deploy backend
# 1. [To Be Avoided]Create a clean requirements file
pip freeze > requirements.txt

# 2. if package and deployment.zip already exists [not needed if requirements.txt hasn't changed]
rm -rf package deployment.zip

# 3. Create a temporary folder for packaging [not needed if requirements.txt hasn't changed]
mkdir package

# 4. Install all your dependencies directly into that folder [not needed if requirements.txt hasn't changed]
pip install -r requirements.txt --target ./package

# 5. Copy your application code and .env file into the folder [not needed if requirements.txt hasn't changed]
cp -r services package/
cp -r common package/
cp main.py package/

# 6. Navigate into the folder and zip everything up!
cd package
zip -r ../deployment.zip .
OR
zip -ur deployment.zip common services main.py

# 7. download the deployment.zip
python -m http.server 8080
